                },
                x: {
                    grid: {
                        display: false
                    }
                }
            },
            plugins: {
                legend: {
                    display: false
                },
                tooltip: {
                    callbacks: {
                        label: function (context) {
                            return context.parsed.y + '% of crashes';
                        }
                    }
                }
            }
        }
    });

    // Leaflet maps
    let turkeyMap, worldMap;

    function createMaps() {
        // Turkey map
        turkeyMap = L.map('turkeyLeafletMap', {
            attributionControl: false
        }).setView([39.0, 35.0], 5);

        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            maxZoom: 18
        }).addTo(turkeyMap);

        const turkeyHotspots = [
            {name: 'Istanbul', coords: [41.0082, 28.9784], value: 100},
            {name: 'Ankara', coords: [39.9334, 32.8597], value: 75},
            {name: 'Izmir', coords: [38.4237, 27.1428], value: 65},
            {name: 'Antalya', coords: [36.8969, 30.7133], value: 60},
            {name: 'Bursa', coords: [40.1950, 29.0600], value: 55},
            {name: 'Gaziantep', coords: [37.0662, 37.3833], value: 50}
        ];

        function getColor(v) {
            if (v > 90) return '#d32f2f';
            if (v > 70) return '#ff6b6b';
            if (v > 55) return '#ff9800';
            return '#4caf50';
        }

        turkeyHotspots.forEach(p => {
            L.circleMarker(p.coords, {
                radius: 8 + p.value / 20,
                fillColor: getColor(p.value),
                color: '#ffffff',
                weight: 1,
                fillOpacity: 0.85
            }).addTo(turkeyMap)
                .bindPopup(`<strong>${p.name}</strong><br>High crash & congestion level`);
        });

        // World map
        worldMap = L.map('worldLeafletMap', {
            attributionControl: false
        }).setView([20, 10], 2);

        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            maxZoom: 18
        }).addTo(worldMap);

        const worldHotspots = [
            {name: 'USA', coords: [37.09, -95.71], value: 75},
            {name: 'Brazil', coords: [-14.23, -51.92], value: 70},
            {name: 'Western Europe', coords: [50, 10], value: 55},
            {name: 'Nigeria', coords: [9.08, 8.68], value: 90},
            {name: 'India', coords: [20.59, 78.96], value: 100},
            {name: 'China', coords: [35.86, 104.19], value: 95},
            {name: 'Thailand', coords: [15.87, 100.99], value: 85},
            {name: 'Australia', coords: [-25.27, 133.77], value: 50}
        ];

        function getWorldColor(v) {
            if (v > 95) return '#d32f2f';
            if (v > 80) return '#ff6b6b';
            if (v > 60) return '#ff9800';
            return '#ffb74d';
        }

        worldHotspots.forEach(p => {
            L.circleMarker(p.coords, {
                radius: 7 + p.value / 25,
                fillColor: getWorldColor(p.value),
                color: '#ffffff',
                weight: 1,
                fillOpacity: 0.85
            }).addTo(worldMap)
                .bindPopup(`<strong>${p.name}</strong><br>High road traffic injury burden`);
        });
    }

    // Tab switching
    function switchTab(evt, tab) {
        const buttons = document.querySelectorAll('.tab-button');
        buttons.forEach(btn => btn.classList.remove('active'));
        evt.currentTarget.classList.add('active');

        document.getElementById('turkey-map').classList.remove('active');
        document.getElementById('world-map').classList.remove('active');
        document.getElementById(tab + '-map').classList.add('active');

        if (tab === 'turkey' && turkeyMap) {
            setTimeout(() => turkeyMap.invalidateSize(), 200);
        }
        if (tab === 'world' && worldMap) {
            setTimeout(() => worldMap.invalidateSize(), 200);
        }
    }

    // Initialize maps
    window.addEventListener('load', function () {
        createMaps();
        if (turkeyMap) {
            setTimeout(() => turkeyMap.invalidateSize(), 200);
        }
    });
